00110 template< typename PT >
00111 void R11AsyncProcess< PT >::SetThreadCount( int_fast8_t threadCount )
00112 {
00113   if( threadCount < 0 || threadCount == threadCount_ )
00114   {
00115     return;
00116   }
00117 
00118   // manually tick until 0
00119   for_each( begin( threads_ ) + currentThread_, end( threads_ ),
00120             [ this ]( R11AsyncProcessThread& thread )
00121             {
00122               thread.Sync();
00123               thread.Resume();
00124             } );
00125 
00126   currentThread_ = 0;
00127 
00128   // sync with all threads
00129   for( auto& thread : threads_ )
00130   {
00131     thread.Sync();
00132   }
00133 
00134   threadCount_ = 0;
00135 
00136   threads_.resize( threadCount );
00137 
00138   // initialize all threads with appropriate callback method
00139   for( uint_fast8_t i = 0; i < threads_.size(); ++i )
00140   {
00141     threads_[ i ].Initialise( std::bind( &PT::Tick, &_process, i ) );
00142   }
00143 
00144   // update _process buffer count to match new thread count
00145   _process.SetBufferCount( threadCount );
00146 
00147   threadCount_ = threadCount;
00148 }

00152 template< typename PT >
00153 void R11AsyncProcess< PT >::Tick()
00154 {
00155   // if multi-threaded, tick the appropriate thread
00156   if( threadCount_ > 0 )
00157   {
00158     threads_[ currentThread_ ].Sync();
00159     threads_[ currentThread_ ].Resume();
00160     ++currentThread_ %= threadCount_;
00161   }
00162   // else tick within current thread
00163   else
00164   {
00165     _process.Tick();
00166   }
00167 }
