00095 template< typename PP >
00096 void R11Process< PP >::SetBufferCount( int_fast8_t bufferCount )
00097 {
00098   if( bufferCount < 0 || bufferCount == _bufferCount )
00099   {
00100     return;
00101   }
00102 
00103   _bufferCount = 0;
00104 
00105   // resize internal buffers
00106   _inputBuffers.resize( bufferCount );
00107   _outputBuffers.resize( bufferCount );
00108 
00109   // resize _gotReleases
00110   _gotReleases.resize( bufferCount );
00111 
00112   if( bufferCount > 0 )
00113   {
00114     // set all gotReleases to false
00115     for( auto& gotRelease : _gotReleases )
00116     {
00117       gotRelease = false;
00118     }
00119 
00120     // set first _gotRelease to true to "kick start" the system
00121     _gotReleases[0] = true;
00122   }
00123 
00124   // reset the mutex and condition variable pointers to new array sizes
00125   _releaseMutexes.reset( new std::mutex[ bufferCount ] );
00126   _releaseCondts.reset( new std::condition_variable_any[ bufferCount ] );
00127 
00128   // update _bufferCount
00129   _bufferCount = bufferCount;
00130 }

00134 template< typename PP >
00135 void R11Process< PP >::Tick( int_fast8_t bufferNo )
00136 {
00137   // if multi-threaded, sync with previous thread before continuing
00138   if( bufferNo >= 0 && bufferNo < _bufferCount )
00139   {
00140     _WaitForRelease( bufferNo );
00141     PP::input_ = _inputBuffers[ bufferNo ];
00142   }
00143 
00144   // call PP's Process_ method
00145   PP::Process_();
00146 
00147   // if multi-threaded, notify the next waiting thread
00148   if( bufferNo >= 0 && bufferNo < _bufferCount )
00149   {
00150     _outputBuffers[ bufferNo ] = PP::output_;
00151     _ReleaseNextBuffer( bufferNo );
00152   }
00153 }

00208 template< typename PP >
00209 void R11Process< PP >::_WaitForRelease( int_fast8_t currentBufferNo )
00210 {
00211   _releaseMutexes[ currentBufferNo ].lock();
00212   if( !_gotReleases[ currentBufferNo ] )
00213   {
00214     _releaseCondts[ currentBufferNo ].wait( _releaseMutexes[ currentBufferNo ] ); // wait for resume
00215   }
00216   _gotReleases[ currentBufferNo ] = false; // reset the release flag
00217   _releaseMutexes[ currentBufferNo ].unlock();
00218 }
